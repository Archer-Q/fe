# Maximum Subarray
`简单`

## JavaScript
```javascript
function maxSubArray(nums) {
  if (!nums || nums.length === 0) return 0

  let res = -Infinity
  let sum = 0
  for (let num of nums) {
    sum = Math.max(sum + num, num)
    res = Math.max(sum, res)
  }
  return res
}
```
