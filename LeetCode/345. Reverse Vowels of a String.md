# Reverse Vowels of a String
`简单`

## JavaScript
```javascript
function reverseVowels(s) {
  let chars = s.split('')
  for (let i = 0, j = chars.length - 1; i < j; ) {
    if (isVowel(s[i])) {
      if (isVowel(s[j])) {
        [chars[i], chars[j]] = [chars[j], chars[i]]
        i += 1
        j -= 1
      } else {
        j -= 1
      }
    } else {
      if (isVowel(s[j])) {
        i += 1
      } else {
        i += 1
        j -= 1
      }
    }
  }
  return chars.join('')

  function isVowel(c) { return /[aeiou]/i.test(c) }
}
```
