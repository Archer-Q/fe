# Count Numbers with Unique Digits

## JavaScript
```javascript
function countNumbersWithUniqueDigits(n) {
  if (n === 0) return 1
  if (n === 1) return 10
  if (n > 10)  return countNumbersWithUniqueDigits(10)
  let avail = 9, count = 9
  for (let i = 0; i + 1 < n; i++) count *= (avail - i)
  return count + countNumbersWithUniqueDigits(n - 1)
}
```
